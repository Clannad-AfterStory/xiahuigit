FIELD
{
    .1139.预计加班起始日期;
    .1139.考勤年月;
}
VAR
{
    String qsrq, S加班起始年月, S加班起始年, S加班起始月;
    int count, total;
}
SERVER
{
    total = FW_GetTotalRecords(1139);
    for(count=0; count<total; count++)
    {
        qsrq = FW_GetValue(1139,预计加班起始日期,count);
        S加班起始年月 = qsrq.Left(7);
        S加班起始年 = S加班起始年月.Left(4);
        S加班起始月 = S加班起始年月.Mid(5);
        S加班起始年月 = S加班起始年 + "年" + S加班起始月 + "月";
    }
    FW_ShowMessage(count);
}
